<pre class="lang:vb decode:true ">Function IsSuji(ByVal strArg As String) As Boolean
    IsSuji = True
    
    For n = 1 To Len(strArg)
       If Asc(Mid(strArg, n, 1)) &lt; 48 Or Asc(Mid(strArg, n, 1)) &gt; 57 Then
            IsSuji = False
            Exit For
        End If
    Next n
    
End Function
