function getEbayDateTo(){
　var now = new Date();
　var y = now.getFullYear();
　var m = now.getMonth() + 1;
　var d = now.getDate();

　if (m &lt; 10) {
  　m = '0' + m;
　}
 
　if (d &lt; 10) {
  　d = '0' + d;
　}

　var new_date = y + '-' + m + '-' + d + 'T00:00:00.000Z';
　return new_date;
}</pre>
