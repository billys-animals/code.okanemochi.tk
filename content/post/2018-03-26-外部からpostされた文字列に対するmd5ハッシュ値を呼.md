```php
&lt;?php

/* 外部からPOST呼出しされた場合の$_POST変数の取得方法 */
$request_body = file_get_contents('php://input');
/* 「$request_body」に対してデコードが必要な場合もある */

$xml=$request_body;
$data = array(
  'message' =&gt; base64_encode(md5($xml, true))
);
header('Content-Type: application/json');
echo json_encode($data);
?&gt;
```
