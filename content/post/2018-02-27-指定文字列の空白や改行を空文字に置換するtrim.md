```js
/************************************************************************************
【関数概要】trim_
　　 
【引数】target

【戻り値】targetの含まれる文字列中の空白を除去した形で呼び出し元に返す
***********************************************************************************/
function trim_(target){
  if (target == null || target == undefined){
    return "";
  }
  return target.replace(/(^\s+)|(\s+$)/g, "");
}
```
